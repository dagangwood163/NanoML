from sklearn.mixture import GMM
from sklearn.metrics import silhouette_score

# Here, I used some ideas and borrowed code from 
# http://scikit-learn.org/stable/auto_examples/mixture/plot_gmm_selection.html
# for the gmm selection.

#Number of possible clusters
v_nb_compn = range(2,6)
#Different gmm types to try to find the best GMM
v_gmm_types = ['spherical','tied','diag','full']
best_gmm = None
highest_sil_score = -np.infty
scores = []
gmms = []

# TODO: Apply your clustering algorithm of choice to the reduced data
# DONE
for gmm_type in v_gmm_types:
    for nb_compn in v_nb_compn:     
        clusterer = GMM(n_components=nb_compn, covariance_type=gmm_type, random_state=12)
        clusterer.fit(reduced_data)
        
        if clusterer.converged_:
            preds = clusterer.predict(reduced_data)
            centers = clusterer.means_
            sample_preds = clusterer.predict(pca_samples)
            scores.append(silhouette_score(reduced_data, preds))
            gmms.append(clusterer)        
            if scores[-1] > highest_sil_score:
                highest_sil_score = scores[-1]
                best_gmm = clusterer

clusterer = best_gmm
print clusterer
print "Has the GMM convedged ? '{}'".format(clusterer.converged_)
